<?php
// Built-in function
$text = "hello world";
echo "Uppercase: " . strtoupper($text) . "<br>";

// User-defined function
function greet($name) {
  return "Hello, $name!";
}
echo greet("Alex") . "<br>";

// Call by value
function swapValue($x, $y) {
  $temp = $x;
  $x = $y;
  $y = $temp;
  echo "Inside swapValue: x=$x, y=$y<br>";
}
$a = 5;
$b = 10;
swapValue($a, $b);
echo "Outside swapValue: a=$a, b=$b<br>";

// Call by reference
function swapReference(&$x, &$y) {
  $temp = $x;
  $x = $y;
  $y = $temp;
}
swapReference($a, $b);
echo "After swapReference: a=$a, b=$b<br>";

// Recursion
function factorial($n) {
  if ($n <= 1) {
    return 1;
  }
  return $n * factorial($n - 1);
}
echo "Factorial of 5: " . factorial(5) . "<br>";
?>
