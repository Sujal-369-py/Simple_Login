<?php
// File inclusion
include 'header.php';
require 'config.php';

// If-Else
$score = 85;
if ($score >= 90) {
  echo "Grade: A<br>";
} elseif ($score >= 75) {
  echo "Grade: B<br>";
} else {
  echo "Grade: C<br>";
}

// Switch
$day = "Tuesday";
switch ($day) {
  case "Monday":
    echo "Start of the week<br>";
    break;
  case "Tuesday":
    echo "Second day<br>";
    break;
  default:
    echo "Midweek or weekend<br>";
}

// Ternary Operator
$age = 20;
$status = ($age >= 18) ? "Adult" : "Minor";
echo "Status: $status<br>";

// For Loop
for ($i = 1; $i <= 5; $i++) {
  echo "For Loop: $i<br>";
}

// While Loop
$j = 1;
while ($j <= 3) {
  echo "While Loop: $j<br>";
  $j++;
}

// Do-While Loop
$k = 1;
do {
  echo "Do-While Loop: $k<br>";
  $k++;
} while ($k <= 3);

// Foreach Loop
$colors = ["Red", "Green", "Blue"];
foreach ($colors as $color) {
  echo "Color: $color<br>";
}
?>
