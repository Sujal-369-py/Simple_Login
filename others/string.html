<?php
$text = "  Hello World!  ";

// Basic functions
echo "Length: " . strlen($text) . "<br>";
echo "Trimmed: " . trim($text) . "<br>";
echo "Uppercase: " . strtoupper($text) . "<br>";
echo "Lowercase: " . strtolower($text) . "<br>";

// Searching and replacing
echo "Position of 'World': " . strpos($text, "World") . "<br>";
echo "Replace 'World' with 'PHP': " . str_replace("World", "PHP", $text) . "<br>";

// Substring and splitting
echo "Substring (0,5): " . substr($text, 0, 5) . "<br>";
$parts = explode(" ", trim($text));
echo "Exploded: " . implode("-", $parts) . "<br>";

// Character-level functions
echo "ASCII of 'A': " . ord("A") . "<br>";
echo "Char from 65: " . chr(65) . "<br>";

// Encoding and hashing
echo "MD5: " . md5($text) . "<br>";
echo "Base64: " . base64_encode($text) . "<br>";

// Count characters
echo "Character count: " . count_chars($text, 3) . "<br>";
?>
