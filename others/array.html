PHP Arrays and Their Types

<?php
// Indexed Array
$fruits = ["Apple", "Banana", "Cherry"];
echo "Second fruit: $fruits[1]<br>";

// Loop through indexed array
foreach ($fruits as $fruit) {
  echo "Fruit: $fruit<br>";
}

// Associative Array
$ages = ["Alex" => 25, "Sam" => 30, "Jamie" => 22];
echo "Sam's age: " . $ages["Sam"] . "<br>";

// Loop through associative array
foreach ($ages as $name => $age) {
  echo "$name is $age years old<br>";
}

// Multidimensional Array
$contacts = [
  ["name" => "Alex", "phone" => "123456"],
  ["name" => "Sam", "phone" => "789012"]
];
echo "Sam's phone: " . $contacts[1]["phone"] . "<br>";

// Array Functions
array_push($fruits, "Mango");
array_pop($fruits);
sort($fruits);
echo "Sorted fruits: " . implode(", ", $fruits) . "<br>";

asort($ages);
ksort($ages);
echo "Sorted ages by key: " . implode(", ", array_values($ages)) . "<br>";

echo "Keys in ages: " . implode(", ", array_keys($ages)) . "<br>";

echo "Is 'Apple' in fruits? " . (in_array("Apple", $fruits) ? "Yes" : "No") . "<br>";

// Explode and Implode
$sentence = "PHP is powerful and flexible";
$words = explode(" ", $sentence);
echo "Words: " . implode(" | ", $words) . "<br>";
?>
