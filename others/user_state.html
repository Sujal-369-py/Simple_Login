<?php
// Start session
session_start();

// Set cookie (expires in 1 hour)
setcookie("username", "Alex", time() + 3600);

// Access cookie
$cookieUser = isset($_COOKIE["username"]) ? $_COOKIE["username"] : "Guest";

// Set session variable
$_SESSION["role"] = "Admin";

// Access session variable
$sessionRole = $_SESSION["role"];

// Current time and expiration
$currentTime = date("Y-m-d H:i:s");
$expireTime = date("Y-m-d H:i:s", time() + 3600);

// Output
echo "Cookie 'username': $cookieUser<br>";
echo "Session 'role': $sessionRole<br>";
echo "Current Time: $currentTime<br>";
echo "Cookie Expires At: $expireTime<br>";
?>
