<?php
// File creation and writing
$file = "demo.txt";
$handle = fopen($file, "w");
fwrite($handle, "Hello, this is a demo file.\nSecond line of text.");
fclose($handle);

// Reading file
if (file_exists($file)) {
  $content = file_get_contents($file);
  echo "File Content:<br>" . $content . "<br>";
}

// Rename file
rename("demo.txt", "renamed.txt");

// Delete file
if (file_exists("renamed.txt")) {
  unlink("renamed.txt");
  echo "File deleted successfully.<br>";
}

// Directory operations
mkdir("demo_folder");
echo "Directory created.<br>";

// List files in directory
$files = scandir("demo_folder");
echo "Files in demo_folder:<br>";
foreach ($files as $f) {
  echo $f . "<br>";
}

// Remove directory
rmdir("demo_folder");
echo "Directory removed.<br>";
?>
